#include <iostream>
using namespace std;

// Definición de la clase
class Persona {
private:
    string nombre;
    int edad;

public:
    // Constructor
    Persona(string _nombre, int _edad) {
        nombre = _nombre;
        edad = _edad;
    }

    // Método para mostrar datos
    void mostrarDatos() {
        cout << "Nombre: " << nombre << ", Edad: " << edad << " años" << endl;
    }

    // Destructor
    ~Persona() {
        cout << "Objeto destruido: " << nombre << endl;
    }
};

// Función principal
int main() {
    // Creación de un objeto
    Persona persona1("Juan", 25);

    // Llamada a un método de la clase
    persona1.mostrarDatos();

    return 0;
}

// Ejecutar en Jupyter
main();
